cmake_minimum_required(VERSION 3.31)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)
project(
    wordle
    VERSION 0.0.0
    LANGUAGES CXX)


set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Gtest Stuff
include(FetchContent)
FetchContent_Declare(
  googletest
  URL https://github.com/google/googletest/archive/refs/tags/v1.17.0.zip
)
# For Windows: Prevent overriding the parent project's compiler/linker settings

set(gtest_force_shared_crt ON CACHE BOOL "" FORCE)
FetchContent_MakeAvailable(googletest)

enable_testing()

file(GLOB_RECURSE TEST_SRC ${PROJECT_SOURCE_DIR}/test/*.cpp)
file(GLOB_RECURSE REGULAR_SRC ${PROJECT_SOURCE_DIR}/srt/*.cpp)
file(GLOB_RECURSE MODULES_SRC ${PROJECT_SOURCE_DIR}/src/*.cppm)

add_executable(wordle-cpp-test)

target_sources(wordle-cpp-test
    PRIVATE
    src/main.cpp
)

target_sources(wordle-cpp-test
    BASE_DIRS
        ${PROJECT_SOURCE_DIR}
    FILE_SET all_my_modules TYPE CXX_MODULES
    FILES
        src/test.cppm
)


target_link_libraries(wordle-cpp-test
    Gtest::gtest_main
    Gtest::gmock
)

include (GoogleTest test)
gtest_discover_tests(wordle-cpp-test)
