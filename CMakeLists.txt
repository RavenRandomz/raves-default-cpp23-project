enable_testing()



message(TRACE ${REGULAR_SRC})

cmake_minimum_required(VERSION 4.2.0)
project(derp CXX)
find_package(GTest REQUIRED)

add_executable(
    module_test
    ${REGULAR_SRC}
    ${MODULES_SRC}
)

target_sources(
    module_test PRIVATE
    FILE_SET all_my_modules TYPE CXX_MODULES
    BASE_DIRS
        ${PROJECT_SOURCE_DIR}
    FILES
    ${MODULES_SRC}
)

list(FILTER TESTING_SRC EXCLUDE REGEX "${PROJECT_SOURCE_DIR}/src/main.cpp" )

add_executable(
    unit_test
    ${TEST_SRC}
    ${TESTING_SRC}
    ${MODULES_SRC}
)

target_sources(
    unit_test PRIVATE
    FILE_SET all_my_modules TYPE CXX_MODULES
    BASE_DIRS
        ${PROJECT_SOURCE_DIR}
    FILES
    ${MODULES_SRC}
)

target_link_libraries(
    unit_test
    GTest::GTest
    GTest::Main
    GTest::gmock
)

include (GoogleTest test)

gtest_discover_tests(unit_test)
